def solution(s):
    answer = []
    
    temp = s[1:-1].split("}")
    for i in range(len(temp)): 
        if i==0: temp[i] = temp[i][1:]
        else: temp[i] = temp[i][2:]

    temp.pop()
    for i in range(len(temp)): 
        temp[i] = list(map(int,temp[i].split(",")))
    
    temp.sort(key = lambda x:len(x))
    
    for chunk in temp:
        for element in chunk:
            if element not in answer: answer.append(element)
    
    return answer
