def solution(board, moves):
    answer = 0
    stack = []
    row, col = len(board), len(board[0])
    
    for cmd in moves:
        cmd -= 1
        '''
        for _ in range(row): print(board[_])
        print("")
        print(stack)
        print("")
        '''
        for i in range(row):
            if board[i][cmd] == 0: continue
            else: 
                stack.append(board[i][cmd])
                board[i][cmd] = 0
                break
        
        if len(stack)>1 and stack[-1] == stack[-2]:
            answer +=1
            stack.pop()
            stack.pop()
    
    return answer*2
