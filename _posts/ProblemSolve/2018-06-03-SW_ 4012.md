        for j in range(len(answer)):
            if i==j: continue
            else: temp.append([answer[i],answer[j]])
    for _ in temp:
        F1+=Map[_[0]][_[1]]

    temp,F2=[],0
    Answer=[i for i in range(len(Map))]
