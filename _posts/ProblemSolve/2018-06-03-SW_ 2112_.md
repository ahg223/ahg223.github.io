def DFS(films,level,start,W,D,K,pos,i):
    global result
    if pos==level:
        a=check(films,W,D,K)
        if a:
            result = 1
            return
    for depth in range(start,D):
        tmp = films[depth]
        films[depth]=[i]*W
        DFS(films,level,depth+1,W,D,K,pos+1,i)
        if result:
            return
        films[depth]=tmp
        
def check(films,W,D,K):
    for row in range(W+1):
        if row == W:
            return True
        else:
            count=1
            before = films[0][row]
            for col in range(1,D+1):
                if col == D:
                    return False
                elif films[col][row]==before:
                    count+=1
                    if count>=K:
                        break
                else:
                    count=1
                    before=films[col][row]
