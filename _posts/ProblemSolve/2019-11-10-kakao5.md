import sys

def solution(stones, k):
    answer = 0
    
    MAX = sys.maxsize
    i, j = 0, k
    
    while True:
        temp =  max(stones[i:j])

        which = stones[i:j].index(temp)
        if MAX > temp: MAX = temp

        which += i
        i, j = which+1, which + k +1

        if j>len(stones): break

        
    return MAX
