Min=0

def check(Map):
    count=0
    for i in range(len(Map)):
        for j in range(len(Map[0])):
            if Map[i][j]!=0: count+=1
    return count

    for j in range(len(Map[0])):
        for i in range(len(Map)):
            if Map[i][j]==0:

                Map[0][j]=0
    return Map
