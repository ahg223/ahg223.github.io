    queue=[[i,j,0]]
    visited=[[False for _  in range(10)] for __ in range(10)]

    while queue!=[]:
        x ,y, length = queue.pop()
        if Map[x][y] == "L": return length - 1
        
        for _ in range(4):
            Nx, Ny = x+dx[_], y+dy[_]

            if not (visited[Ny][Nx] or Map[Ny][Nx]=="R"): continue

print(bfs(i, j))
